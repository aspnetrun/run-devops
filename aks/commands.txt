Install the Azure CLI
	https://docs.microsoft.com/en-us/cli/azure/install-azure-cli
	https://docs.microsoft.com/en-us/cli/azure/install-azure-cli-windows?tabs=azure-cli
--
az version

{
  "azure-cli": "2.16.0",
  "azure-cli-core": "2.16.0",
  "azure-cli-telemetry": "1.0.6",
  "extensions": {}
}
--
az login
--
Create a resource group
az group create --name myResourceGroup --location westeurope
--
Create an Azure Container Registry
az acr create --resource-group myResourceGroup --name swnzenacr --sku Basic
--
Enable Admin Account for ACR Pull
az acr update -n swnzenacr --admin-enabled true
--
Log in to the container registry
az acr login --name swnzenacr
Login Succeeded
--
Tag a container image

get the login server address
az acr list --resource-group myResourceGroup --query "[].{acrLoginServer:loginServer}" --output table
swnzenacr.azurecr.io
--
Tag your images

docker tag shoppingapi:latest swnzenacr.azurecr.io/shoppingapi:v1
docker tag shoppingclient:latest swnzenacr.azurecr.io/shoppingclient:v1

Check
docker images
--
Push images to registry

docker push swnzenacr.azurecr.io/shoppingapi:v1
docker push swnzenacr.azurecr.io/shoppingclient:v1
--
List images in registry
az acr repository list --name swnzenacr --output table

Result
shoppingapi
shoppingclient
--
See tags
az acr repository show-tags --name swnzenacr --repository shoppingclient --output table

Result
v1
--
Create AKS cluster with attaching ACR
az aks create --resource-group myResourceGroup --name myAKSCluster --node-count 1 --generate-ssh-keys --attach-acr swnzenacr

--
Install the Kubernetes CLI
az aks install-cli
--
Connect to cluster using kubectl
az aks get-credentials --resource-group myResourceGroup --name myAKSCluster

To verify;
kubectl get nodes
kubectl get all
--
Check Kube Config

kubectl config get-contexts
kubectl config current-context
kubectl config use-context gcpcluster-k8s-1
	Switched to context "gcpcluster-k8s-1"
